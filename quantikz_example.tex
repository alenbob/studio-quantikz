\section*{Quantikz Examples}

\[
\begin{quantikz}
\lstick{\ket{0}} & \gate{H} & \gate{Z} & \meter{}
\end{quantikz}
\]

\[
\begin{quantikz}
\lstick{\ket{0}} & \gate{H} & \ctrl{1} & \meter{} \\
\lstick{\ket{0}} &          & \targ{}  & \meter{}
\end{quantikz}
\]

\[
\begin{quantikz}
& \ctrl{1} & \qquad & & \ctrl{1} & \\
& \ctrl{1} & \qquad & & \ctrl{1} & \\
& \wireoverride{n} & \qquad & & & \wireoverride{n}
\end{quantikz}
\]

\[
\begin{quantikz}
\lstick{\ket{\psi}} & \gate[2]{U} & \swap{1} & \meter{} \\
\lstick{\ket{0}}    &             & \targX{} & \meter{}
\end{quantikz}
\]

\[
\begin{quantikz}

\end{quantikz}
\]

\[
\begin{quantikz}
